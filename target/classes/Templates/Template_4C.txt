#TEMPLATE 4C
SELECT ?ticker ?valor
WHERE {
    {
        SELECT ?ticker ?valor WHERE {
            #FILTER_BLOCK#
            ?S1 P1 ?SO1 .
            ?SO1 P8 ?ticker .
            ?SO1 P3 ?negociadoNode .
            ?negociadoNode P5 ?S2 .
            ?S2 P6 "#DATA#"^^xsd:date .
            ?negociadoNode #VALOR_DESEJADO# ?valor .
        }
    }
    UNION
    {
        SELECT ("TOTAL" AS ?ticker) (SUM(?soma_valor) AS ?valor) WHERE {
            #FILTER_BLOCK#
            ?S1 P1 ?SO1 .
            ?SO1 P3 ?negociadoNode .
            ?negociadoNode P5 ?S2 .
            ?S2 P6 "#DATA#"^^xsd:date .
            ?negociadoNode #VALOR_DESEJADO# ?soma_valor .
        }
    }
}
