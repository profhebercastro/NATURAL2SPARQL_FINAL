# Finalidade: Verifica se uma empresa pertence a um determinado setor.
# Exemplo: "A Gerdau atua no setor de mineração?"
# ----------------------------------------------------------
ASK WHERE {
    ?S1 P7 ?label .
    FILTER(REGEX(STR(?label), "#ENTIDADE_NOME#", "i"))
    
    ?S1 P9 ?setor .
    ?setor P7 ?setorLabel .
    FILTER(REGEX(STR(?setorLabel), "#NOME_SETOR#", "i"))
}
