# Finalidade: Busca o ticker, o nome da empresa e o seu setor de atuação a partir do NOME da empresa.
# Exemplo: "Qual o ticker e o setor da Gerdau?"
# ----------------------------------------------------------
SELECT DISTINCT ?ticker ?empresa ?setorEmpresa WHERE {
  
    ?S1 P7 ?empresa .
    FILTER(REGEX(STR(?empresa), "#ENTIDADE_NOME#", "i"))
    ?S1 P1 ?SO1 .
    ?SO1 P8 ?ticker .
   
    ?S1 P9 ?setor .
   
    ?setor P7 ?setorEmpresa .
}
ORDER BY ?empresa
