# Finalidade: Busca o(s) ticker(s) de uma empresa pelo NOME.
# Exemplo: "Qual o código de negociação da Vale?"
# ----------------------------------------------------------
SELECT DISTINCT ?ticker WHERE {
    ?S1 P7 ?label .
    FILTER(REGEX(STR(?label), "#ENTIDADE_NOME#", "i"))
    ?S1 P1 ?SO1 .
    ?SO1 P8 ?ticker .
}
ORDER BY ?ticker
