# Qual foi o preço máximo da ação ordinária da Gerdau em 08/05/2023?
SELECT ?ANS WHERE {
    ?S1 P7 "#ENTIDADE_NOME#"@pt .
    ?S1 P1 ?SO1 .
    ?SO1 P2 ?O1 .                      
    ?SO1 P3 ?SO2 .
    ?SO2 P5 ?S2 .
    ?S2 P6 "#DATA#"^^xsd:date .
    ?SO2 #VALOR_DESEJADO# ?ANS .

    FILTER(REGEX(STR(?O1), "#REGEX_PATTERN#"))
}
LIMIT 1
